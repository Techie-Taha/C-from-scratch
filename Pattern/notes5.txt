// Rectangle

#include<iostream>
using namespace std;
int main(){

  
    int m, n;
    cin >> m; // m is the number of rows
    cin >> n; // n is the number of columns
    for( int i = 1; i <= m; i++){ // This line takes the rows
        for(int j = 1; j <= n; j++){ // This lines takes the columns
            cout << "* "; 
        }
        cout << endl;
    }

return 0;
}
// Line 13 will only start  when the condition in line 12 is true. Once the condition in line 12 is true, the loop starts from 12 to 16, after that the value of i increases to 2.
